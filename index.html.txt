<script>
  const messages = document.getElementById('messages');
  const userInput = document.getElementById('userInput');
  const sendBtn = document.getElementById('sendBtn');

  function addMessage(text, fromUser = true) {
    const div = document.createElement('div');
    div.textContent = (fromUser ? 'Você: ' : 'ChatBot AI: ') + text;
    div.style.marginBottom = '8px';
    div.style.whiteSpace = 'pre-wrap';
    messages.appendChild(div);
    messages.scrollTop = messages.scrollHeight;
  }

  function chatbotResponse(input) {
    const lower = input.toLowerCase();
    if (lower.includes('oi') || lower.includes('olá')) {
      return 'Olá! Como posso ajudar você hoje?';
    }
    if (lower.includes('minecraft')) {
      return 'Posso ajudar a criar mods, mapas e shaders para Minecraft PE!';
    }
    if (lower.includes('ajuda')) {
      return 'Claro! Pergunte o que quiser sobre o ChatBot AI.';
    }
    if (lower.trim() === '') {
      return 'Por favor, digite alguma coisa.';
    }
    return 'Desculpe, ainda estou aprendendo. Pode tentar outra coisa?';
  }

  sendBtn.addEventListener('click', () => {
    const input = userInput.value.trim();
    if (!input) return;
    addMessage(input, true);
    userInput.value = '';
    setTimeout(() => {
      addMessage(chatbotResponse(input), false);
    }, 800);
  });

  userInput.addEventListener('keypress', e => {
    if (e.key === 'Enter') sendBtn.click();
  });
</script>
